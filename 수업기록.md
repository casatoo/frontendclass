```js
let order = 0;

$(".btns > button")
  .eq(0)
  .on("click", function () {
    if (order == 0) {
      order = lastOrder;
    } else {
      order--;
    }
    $(".slides>div").removeClass("active");

    $(".slides>div").eq(order).addClass("active");
  });

$(".btns > button")
  .eq(1)
  .on("click", function () {
    if (order == lastOrder) {
      order = 0;
    } else {
      order++;
    }
    $(".slides>div").removeClass("active");

    $(".slides>div").eq(order).addClass("active");
  });
```
